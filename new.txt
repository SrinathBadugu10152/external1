def create_sample_file():
    try:
        # Open the file in write mode to create it
        with open("sample.txt", 'w') as file:
            # Write some content to the file
            file.write("This is a sample file.\nIt contains some text.\n")

        print("File 'sample.txt' created successfully.")
    except Exception as e:
        print("An error occurred while creating the file:", str(e))

def count_characters():
    try:
        # Open the file in read mode
        with open("sample.txt", 'r') as file:
            # Read the content of the file
            content = file.read()

        # Count the number of characters in the content
        num_characters = len(content)

        print("Number of characters in the file:", num_characters)
    except FileNotFoundError:
        print("The file 'sample.txt' does not exist.")

# Create the sample file
create_sample_file()

# Count the number of characters in the file
count_characters()
